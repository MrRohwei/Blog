### 2. 博客详情（BlogDetailView.vue）— 重点页面

```vue
<template>
  <div class="detail-layout">
    <article class="article">
      <h1>{{ blog.title }}</h1>
      <div class="meta">
        <span>{{ blog.createdAt }}</span>
        <span>{{ blog.views }} 次阅读</span>
        <span>{{ blog.categoryName }}</span>
      </div>
      <!-- Markdown 渲染区 -->
      <div class="markdown-body" v-html="renderedContent"></div>

      <!-- 标签 + 点赞 -->
      <div class="blog-footer">
        <el-tag v-for="tag in blog.tags" :key="tag.id" :style="{ backgroundColor: tag.color }">
          {{ tag.name }}
        </el-tag>
        <LikeButton :blogId="blog.id" :count="blog.likes" />
      </div>

      <!-- 底部广告位 -->
      <AdvertisementWidget position="DETAIL_BOTTOM" />

      <!-- 评论区 -->
      <CommentSection :blogId="blog.id" />
    </article>

    <!-- 目录 TOC -->
    <aside class="toc-sidebar">
      <TableOfContents :content="renderedContent" />
    </aside>
  </div>
</template>

<script setup>
import { ref, computed, onMounted } from 'vue'
import { useRoute } from 'vue-router'
import { marked } from 'marked'
import hljs from 'highlight.js'
import DOMPurify from 'dompurify'
import 'highlight.js/styles/github.css'
import { getBlogDetail } from '@/api/blog'

const route = useRoute()
const blog = ref({})

// 配置 marked 代码高亮
marked.setOptions({
  highlight: (code, lang) => {
    const language = hljs.getLanguage(lang) ? lang : 'plaintext'
    return hljs.highlight(code, { language }).value
  },
  langPrefix: 'hljs language-'
})

const renderedContent = computed(() => {
  if (!blog.value.content) return ''
  return DOMPurify.sanitize(marked(blog.value.content))
})

onMounted(async () => {
  blog.value = await getBlogDetail(route.params.id)
  // 动态设置页面标题
  document.title = `${blog.value.title} - Voidpen`
})
</script>
```

### 3. 目录导航组件（TableOfContents.vue）

```javascript
// 从渲染后的 HTML 中提取 h2/h3 标题
function extractHeadings(htmlContent) {
  const parser = new DOMParser()
  const doc = parser.parseFromString(htmlContent, 'text/html')
  const headings = doc.querySelectorAll('h2, h3')
  return Array.from(headings).map((h, i) => ({
    id: `heading-${i}`,
    text: h.textContent,
    level: parseInt(h.tagName[1])
  }))
}

// 滚动监听，高亮当前所在标题
onMounted(() => {
  const observer = new IntersectionObserver(entries => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        activeId.value = entry.target.id
      }
    })
  }, { rootMargin: '-20% 0px -70% 0px' })

  headings.value.forEach(h => {
    const el = document.getElementById(h.id)
    if (el) observer.observe(el)
  })
})
```

### 5. 点赞按钮组件（LikeButton.vue）

```vue
<script setup>
import { ref } from 'vue'
import { likeBlog } from '@/api/blog'

const props = defineProps({ blogId: Number, count: Number })
const liked = ref(localStorage.getItem(`liked_${props.blogId}`) === '1')
const count = ref(props.count)

async function handleLike() {
  if (liked.value) return
  await likeBlog(props.blogId)
  liked.value = true
  count.value++
  localStorage.setItem(`liked_${props.blogId}`, '1')
}
</script>
```

### 6. 轮播图组件（BannerSwiper.vue）

使用 CSS 实现自动轮播（不引入第三方 Swiper 库，减少包体积）：

```javascript
// 每 4 秒切换，鼠标 hover 暂停
const currentIndex = ref(0)
let timer = null

function startAutoPlay() {
  timer = setInterval(() => {
    currentIndex.value = (currentIndex.value + 1) % props.banners.length
  }, 4000)
}

onMounted(startAutoPlay)
onUnmounted(() => clearInterval(timer))
```
